    ```typescript
    import { create } from 'zustand';
    import { temporal } from 'zundo';
    import type { Circuit, Component, Wire, ComponentType } from '@/circuit/types';

    interface CircuitState {
      circuit: Circuit;
      refCounters: Record<string, number>; // { R: 3, C: 1, V: 2 } for auto-naming

      // Actions
      addComponent: (type: ComponentType, position: { x: number; y: number }) => string; // returns component ID
      removeComponent: (id: string) => void;
      updateComponentValue: (id: string, value: string) => void;
      updateComponentPosition: (id: string, position: { x: number; y: number }) => void;
      rotateComponent: (id: string) => void; // cycles 0 -> 90 -> 180 -> 270
      addWire: (sourcePortId: string, targetPortId: string) => string; // returns wire ID
      removeWire: (id: string) => void;
      clearCircuit: () => void;
      // Bulk operations for copy/paste
      addComponents: (components: Component[]) => void;
    }

    export const useCircuitStore = create<CircuitState>()(
      temporal(
        (set, get) => ({
          circuit: { components: new Map(), wires: new Map(), nets: new Map() },
          refCounters: {},
          // ... implement all actions
        }),
        {
          limit: 100,
          partialize: (state) => ({
            circuit: state.circuit,
            refCounters: state.refCounters,
          }),
        }
      )
    );
    ```

    Key implementation details:
    - addComponent generates ID via crypto.randomUUID(), creates refDesignator from spicePrefix + counter (R1, R2, C1...), creates Port objects from ComponentDefinition.ports, uses COMPONENT_LIBRARY for defaults
    - removeComponent also removes any wires connected to the component's ports
    - rotateComponent cycles: 0 -> 90 -> 180 -> 270 -> 0
    - Only circuit data and refCounters are tracked by zundo (not actions)
    - Undo/redo accessed via useCircuitStore.temporal.getState().undo() / .redo()
