    export type ComponentType =
      | 'resistor' | 'capacitor' | 'inductor' | 'transformer'

      type: ComponentType;
      refDesignator: string;  // "R1", "C2", "V1"

    }

      wires: Map<string, Wire>;
      nets: Map<string, Net>;
    }

    export interface AnalysisConfig {

    Create src/circuit/validator.ts implementing validateCircuit per Research Pattern:

    ```typescript
    export interface ValidationError {
      type: 'floating_node' | 'no_ground' | 'source_loop' | 'disconnected';
      message: string;
      suggestion: string;
      componentIds: string[];
      severity: 'error' | 'warning';
    }

    export function validateCircuit(circuit: Circuit): ValidationError[]
    ```

    Validation checks (per D-22, ERR-03):
    1. No ground: Check if any component has type === 'ground'. Error if missing.
    2. Floating nodes: For each component, check each port's netId. Warning if null (unconnected).
    3. Voltage source loops: Build a subgraph of only voltage sources and check for cycles where two voltage sources share both nets.
    4. Disconnected components: Warning if ALL ports of a component are unconnected.

    Message text per UI-SPEC Copywriting Contract:
    - No ground: "No ground connection" with suggestion "Every node needs a path to ground (node 0). Add a ground symbol to your circuit."
    - Floating node: "{refDesignator} pin \"{portName}\" is not connected." with suggestion "Connect this pin to complete the circuit."
    - Source loop: "Voltage source loop between {V1} and {V2}." with suggestion "Remove one voltage source or add a resistor between them."

    Create src/simulation/errorTranslator.ts implementing translateError per Research Pattern 6:

    ```typescript
    export interface TranslatedError {
      message: string;
      suggestion: string;
      componentRef?: string;
      severity: 'error' | 'warning';
      raw: string;
    }

    export function translateError(raw: string, netMap?: Map<string, string>): TranslatedError
    ```

    Error patterns to match (per ERR-01, ERR-02, ERR-04):
    1. /singular matrix.*node (\w+)/i -- floating node / short circuit
    2. /no dc path to ground.*node (\w+)/i -- missing ground path
    3. /timestep too small/i -- convergence failure
    4. /can't find model (\w+)/i -- missing SPICE model (ERR-04)
    5. /\b([RCLVDQMXI]\d+)\b/ -- extract component reference from any error

    Fallback: if no pattern matches, return generic error with raw output preserved.
