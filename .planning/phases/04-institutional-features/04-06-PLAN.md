    STEP 2 — `src/report/katexRasterize.ts`. Pre-rasterize per 04-01 spike decision (default: PNG via html-to-image; fallback: SVG embed if spike failed):
    ```typescript
    import katex from 'katex';
    import 'katex/dist/katex.min.css';
    import { toPng } from 'html-to-image';

    /**
     * Render a LaTeX formula to a PNG data URL.
     * Awaits document.fonts.ready first (Pitfall 2).
     * Uses an offscreen absolutely-positioned div to avoid layout shift.
     */
    export async function rasterizeKatex(
      tex: string,
      options: { displayMode?: boolean; pixelRatio?: number } = {},
    ): Promise<string> {
      await document.fonts.ready;
      const host = document.createElement('div');
      host.style.position = 'fixed';
      host.style.left = '-10000px';
      host.style.top = '0';
      host.style.padding = '4px 8px';
      host.style.background = '#ffffff';
      host.style.color = '#000000';
      document.body.appendChild(host);
      try {
        katex.render(tex, host, {
          throwOnError: false,
          displayMode: options.displayMode ?? false,
          output: 'html',
        });
        // Let layout settle so html-to-image measures correctly
        await new Promise(r => requestAnimationFrame(r));
        return await toPng(host, { pixelRatio: options.pixelRatio ?? 2, backgroundColor: '#ffffff' });
      } finally {
        document.body.removeChild(host);
      }
    }

    /**
     * Rasterize multiple formulas in parallel.
     */
    export async function rasterizeKatexBatch(formulas: string[]): Promise<string[]>;
    ```

    /**
     * Compose ReportData from current circuit + simulation state.
     * Called by ReportPreviewPage before rendering ReportLayout.

    export async function buildReportData(opts: {
      circuitElement: HTMLElement;           // React Flow canvas element to screenshot
      waveformCanvases: HTMLCanvasElement[]; // uPlot canvases
      title: string;
      author: string;
      measurements: Array<{ label: string; tex: string; value?: string }>;
      annotations: string;
      course?: string;
      labCheckpoints?: ReportData['labCheckpoints'];
    }): Promise<ReportData> {
      const paperSize: 'letter' | 'a4' =
        navigator.language.startsWith('en-US') ? 'letter' : 'a4';
      const schematicPngDataUrl = await exportSchematicAsPng(opts.circuitElement);
      const waveforms = opts.waveformCanvases.map((c, i) => ({
        label: `Waveform ${i + 1}`,
        pngDataUrl: c.toDataURL('image/png'),
      }));
      const measurementPngs = await rasterizeKatexBatch(opts.measurements.map(m => m.tex));
      return {
        title: opts.title,
        author: opts.author,
        date: new Date().toISOString(),
        course: opts.course,
        schematicPngDataUrl,
        waveforms,
        measurements: opts.measurements.map((m, i) => ({ ...m, pngDataUrl: measurementPngs[i] })),
        annotations: opts.annotations,
        labCheckpoints: opts.labCheckpoints,
        paperSize,
      };
    }

    export async function exportReportAsPdf(
      element: HTMLElement,
      filename: string,
      paperSize: 'letter' | 'a4' = 'letter',
    ): Promise<void> {
      await document.fonts.ready;
      const doc = new jsPDF({
        unit: 'px',
        format: paperSize,
        hotfixes: ['px_scaling'],
        orientation: 'portrait',
      });
      await doc.html(element, {
        autoPaging: 'text',
        margin: [40, 40, 40, 40],
        width: paperSize === 'letter' ? 612 : 595,
        windowWidth: element.scrollWidth,
        html2canvas: {
          scale: 2,
          backgroundColor: '#ffffff',
          useCORS: true,
        },
      });
      doc.save(filename);
    }

    const LATEX_ESCAPES: Record<string, string> = {
      '&': '\\&', '%': '\\%', '$': '\\$', '#': '\\#',
      '_': '\\_', '{': '\\{', '}': '\\}',
      '~': '\\textasciitilde{}', '^': '\\textasciicircum{}',
      '\\': '\\textbackslash{}',
    };

    export function escapeLatex(s: string): string {
      return s.replace(/[&%$#_{}~^\\]/g, ch => LATEX_ESCAPES[ch] ?? ch);
    }

    export function renderLatex(data: ReportData): string {
      // Template literal .tex source, using escapeLatex on all user fields
      // and including measurements in math mode with raw tex (NOT escaped — it's already LaTeX)
    }

    export async function exportReportAsLatexZip(data: ReportData): Promise<void> {
      const blob = await buildLatexZipBlob(data);
      const url = URL.createObjectURL(blob);
      const a = document.createElement('a');
      a.href = url;
      a.download = `${data.title.replace(/\W+/g, '-')}.zip`;
      a.click();
      URL.revokeObjectURL(url);
    }

    export async function buildLatexZipBlob(data: ReportData): Promise<Blob> {
      const zip = new JSZip();
      zip.file('report.tex', renderLatex(data));
      const figures = zip.folder('figures')!;
      figures.file('schematic.png', dataUrlToUint8(data.schematicPngDataUrl));
      data.waveforms.forEach((w, i) => {
        figures.file(`waveform-${i + 1}.png`, dataUrlToUint8(w.pngDataUrl));
      });
      data.measurements.forEach((m, i) => {
        figures.file(`measurement-${i + 1}.png`, dataUrlToUint8(m.pngDataUrl));
      });
      zip.file('README.md', '# Lab report\n\nCompile with `pdflatex report.tex`.\n');
      return zip.generateAsync({ type: 'blob' });
    }
