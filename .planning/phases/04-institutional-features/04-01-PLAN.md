    ```typescript
    import { SignJWT, importPKCS8 } from 'jose';
    import { readFileSync } from 'fs';
    import { resolve } from 'path';

    const privatePem = readFileSync(
      resolve(__dirname, '../fixtures/lti/mock-platform-private.pem'),
      'utf8',
    );

    /**
     * Sign an unsigned LTI claim fixture with the mock platform's private key.
     * Used by verify.test.ts, oidc.test.ts, deepLink.test.ts — anywhere we need
     * a real JWT that `jose.jwtVerify` can validate against jwks.json.
     */
    export async function signFixtureIdToken(
      claims: Record<string, unknown>,
      overrides: Partial<{ exp: number; iat: number; nonce: string }> = {},
    ): Promise<string> {
      const pk = await importPKCS8(privatePem, 'RS256');
      const now = Math.floor(Date.now() / 1000);
      return await new SignJWT({
        ...claims,
        iat: overrides.iat ?? now,
        exp: overrides.exp ?? now + 600,
        nonce: overrides.nonce ?? String(claims.nonce ?? crypto.randomUUID()),
      })
        .setProtectedHeader({ alg: 'RS256', kid: 'mock-platform-2026', typ: 'JWT' })
        .sign(pk);
    }
    ```

    ```typescript
    import { readFileSync } from 'fs';
    import { resolve } from 'path';

    /**
     * In-memory LTI platform mock.
     * - Serves JWKS from fixtures
     * - Simulates token endpoint (client_credentials grant) for AGS tests
     * - Records score POSTs for assertions
     */
    export function createMockPlatform() {
      const jwks = JSON.parse(
        readFileSync(resolve(__dirname, '../fixtures/lti/jwks.json'), 'utf8'),
      );
      const tokenIssued: Array<{ assertion: string; scope: string }> = [];
      const scoresReceived: Array<{ lineItemUrl: string; contentType: string; body: unknown }> = [];

      return {
        jwks,
        async fetch(input: Request | string, init?: RequestInit): Promise<Response> {
          const url = new URL(typeof input === 'string' ? input : input.url);
          if (url.pathname.endsWith('/.well-known/jwks.json')) {
            return new Response(JSON.stringify(jwks), { headers: { 'Content-Type': 'application/json' } });
          }
          if (url.pathname.endsWith('/login/oauth2/token')) {
            const body = new URLSearchParams(init?.body as string);
            tokenIssued.push({ assertion: body.get('client_assertion') ?? '', scope: body.get('scope') ?? '' });
            return new Response(JSON.stringify({ access_token: 'mock-bearer', expires_in: 3600 }), {
              headers: { 'Content-Type': 'application/json' },
            });
          }
          if (url.pathname.endsWith('/scores')) {
            scoresReceived.push({
              lineItemUrl: url.href.replace(/\/scores$/, ''),
              contentType: (init?.headers as Record<string, string>)?.['Content-Type'] ?? '',
              body: JSON.parse(init?.body as string),
            });
            return new Response('', { status: 204 });
    ```
