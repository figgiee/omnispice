    ```typescript
    /**
     * Linear interpolation: value of `y` at time `t` given sample points (ts, ys).
     * Clamps to endpoints if t is outside [ts[0], ts[last]].
     */
    export function interpAt(ts: Float64Array, ys: Float64Array, t: number): number;

    /**
     * Resample `y` onto a new time grid `newTs` via linear interpolation.
     */
    export function resample(ts: Float64Array, ys: Float64Array, newTs: Float64Array): Float64Array;

    /**
     * Root-mean-square error between student and reference waveforms.
     * Interpolates reference onto student's time grid first so mismatched grids are handled.
     * Returns Infinity if student is empty.
     */
    export function rmse(
      studentTime: Float64Array,
      studentValues: Float64Array,
      refTime: Float64Array,
      refValues: Float64Array,
    ): number;

    /**
     * Max absolute difference between student and reference (same interpolation strategy as rmse).
     */
    export function maxAbs(
      studentTime: Float64Array,
      studentValues: Float64Array,
      refTime: Float64Array,
      refValues: Float64Array,
    ): number;

    /**
     * Check whether a single scalar value is within tolerance of expected.
     * tol_pct = percent of |expect|; tol_abs = absolute magnitude.
     */
    export function withinTolerance(actual: number, expected: number, tol: { tol_pct: number } | { tol_abs: number }): boolean;
    ```
