    ```typescript
    import { useUser, useAuth } from '@clerk/clerk-react';
    import { useQueryClient } from '@tanstack/react-query';

    export type Role = 'instructor' | 'student';

    /**
     * Read the current user's role from Clerk publicMetadata.
     * Defaults to 'student' when undefined (pilot trust model per D-01).
     */
    export function useRole(): Role {
      const { user } = useUser();
      const raw = user?.publicMetadata?.role;
      return raw === 'instructor' ? 'instructor' : 'student';
    }

    /**
     * Flip the current user's role to 'instructor' via Worker endpoint,
     * then force-refresh the Clerk JWT to pick up the new custom claim (Pitfall 1).
     */
    export function useBecomeInstructor() {
      const { user } = useUser();
      const { getToken } = useAuth();
      const queryClient = useQueryClient();
      const apiBase = import.meta.env.VITE_API_URL ?? 'http://localhost:8787';

      return async (): Promise<void> => {
        const token = await getToken();
        const res = await fetch(`${apiBase}/api/me/become-instructor`, {
          method: 'POST',
          headers: {
            Authorization: `Bearer ${token}`,
            'Content-Type': 'application/json',
          },
        });
        if (!res.ok) {
          throw new Error(`Failed to become instructor: ${res.status}`);
        }
        // Reload publicMetadata from Clerk and force a fresh JWT (Pitfall 1).
        await user?.reload();
        await getToken({ skipCache: true });
        // Invalidate classroom-dependent queries so dashboard re-renders as instructor.
        queryClient.invalidateQueries({ queryKey: ['courses'] });
      };
    }
    ```

    import type {
      Course, CourseDetail, AssignmentDetail, Submission, SubmissionListRow,
      CreateCourseInput, CreateCourseResponse,
      JoinCourseInput, JoinCourseResponse,
      CreateAssignmentInput,
      SubmitAssignmentResponse,
      SaveGradeInput,

    const API_BASE = import.meta.env.VITE_API_URL ?? 'http://localhost:8787';

    async function authedFetch(
      path: string,
      options: RequestInit,
      getToken: () => Promise<string | null>
    ): Promise<Response> {
      const token = await getToken();
      const headers = new Headers(options.headers ?? {});
      if (token) headers.set('Authorization', `Bearer ${token}`);
      if (options.body && !headers.has('Content-Type')) {
        headers.set('Content-Type', 'application/json');
      }
      const res = await fetch(`${API_BASE}${path}`, { ...options, headers });
      if (!res.ok) {
        const body = await res.text();
        throw new Error(`API ${options.method ?? 'GET'} ${path} → ${res.status}: ${body}`);
      }
      return res;
    }

    type TokenFn = () => Promise<string | null>;

    export async function listCourses(getToken: TokenFn): Promise<Course[]> {
      const res = await authedFetch('/api/courses', { method: 'GET' }, getToken);
      return res.json() as Promise<Course[]>;
    }

    export async function createCourse(input: CreateCourseInput, getToken: TokenFn): Promise<CreateCourseResponse> {
      const res = await authedFetch('/api/courses', { method: 'POST', body: JSON.stringify(input) }, getToken);
      return res.json() as Promise<CreateCourseResponse>;
    }

    export async function getCourse(courseId: string, getToken: TokenFn): Promise<CourseDetail> {
      const res = await authedFetch(`/api/courses/${courseId}`, { method: 'GET' }, getToken);
      return res.json() as Promise<CourseDetail>;

      const res = await authedFetch('/api/courses/join', {

    import { useMutation, useQuery, useQueryClient } from '@tanstack/react-query';
    import { useCurrentUser } from '@/auth/useCurrentUser';
    import {
      listCourses, createCourse, getCourse, deleteCourse, joinCourse,
      createAssignment, getAssignment, updateAssignment, deleteAssignment,
      submitAssignment, listSubmissions, getMySubmission,
      getSubmission, saveGrade,
    } from './classroomApi';
    import type {
      CreateCourseInput, JoinCourseInput, CreateAssignmentInput,
      SaveGradeInput, SubmissionListRow,
    } from './classroomTypes';

    // ===== Queries (D-31 key hierarchy) =====

    export function useCourses() {
      const { getToken, isSignedIn } = useCurrentUser();
      return useQuery({
        queryKey: ['courses'],
        queryFn: () => listCourses(getToken),
        enabled: isSignedIn,
        staleTime: 30_000,
      });
    }

    export function useCourse(courseId: string | null) {
      const { getToken } = useCurrentUser();
      return useQuery({
        queryKey: ['course', courseId],
        queryFn: () => getCourse(courseId!, getToken),
        enabled: !!courseId,
      });
    }

    export function useAssignment(assignmentId: string | null) {
      const { getToken } = useCurrentUser();
      return useQuery({
        queryKey: ['assignment', assignmentId],
        queryFn: () => getAssignment(assignmentId!, getToken),
        enabled: !!assignmentId,
      });
    }

    export function useSubmissions(assignmentId: string | null) {
      const { getToken } = useCurrentUser();
      return useQuery({
        queryKey: ['assignment', assignmentId, 'submissions'],
        queryFn: () => listSubmissions(assignmentId!, getToken),
        enabled: !!assignmentId,
      });
    }

    export function useMySubmission(assignmentId: string | null) {
      const { getToken, isSignedIn } = useCurrentUser();
      return useQuery({
        queryKey: ['mySubmission', assignmentId],
        queryFn: () => getMySubmission(assignmentId!, getToken),
        enabled: isSignedIn && !!assignmentId,
      });
    }

    export function useSubmission(submissionId: string | null) {
      const { getToken } = useCurrentUser();
      return useQuery({
        queryKey: ['submission', submissionId],
        queryFn: () => getSubmission(submissionId!, getToken),
        enabled: !!submissionId,
      });
    }

    // ===== Mutations (invalidation map per 03-RESEARCH.md) =====

    export function useCreateCourse() {
      const { getToken } = useCurrentUser();
      const qc = useQueryClient();
      return useMutation({
        mutationFn: (input: CreateCourseInput) => createCourse(input, getToken),
        onSuccess: () => qc.invalidateQueries({ queryKey: ['courses'] }),
      });
    }

    export function useDeleteCourse() {
      const { getToken } = useCurrentUser();
      const qc = useQueryClient();
      return useMutation({
        mutationFn: (courseId: string) => deleteCourse(courseId, getToken),
        onSuccess: (_r, courseId) => {
          qc.invalidateQueries({ queryKey: ['courses'] });
          qc.invalidateQueries({ queryKey: ['course', courseId] });
        },
      });
    }

    export function useJoinCourse() {
      const { getToken } = useCurrentUser();
      const qc = useQueryClient();
      return useMutation({
        mutationFn: (input: JoinCourseInput) => joinCourse(input, getToken),
        onSuccess: () => qc.invalidateQueries({ queryKey: ['courses'] }),
      });
    }

    export function useCreateAssignment(courseId: string) {
      const { getToken } = useCurrentUser();
      const qc = useQueryClient();
      return useMutation({
        mutationFn: (input: CreateAssignmentInput) => createAssignment(courseId, input, getToken),
        onSuccess: () => qc.invalidateQueries({ queryKey: ['course', courseId] }),
      });
    }

    export function useUpdateAssignment(assignmentId: string, courseId: string) {
      const { getToken } = useCurrentUser();
      const qc = useQueryClient();
      return useMutation({
        mutationFn: (input: Partial<CreateAssignmentInput>) =>
          updateAssignment(assignmentId, input, getToken),
        onSuccess: () => {
          qc.invalidateQueries({ queryKey: ['course', courseId] });
          qc.invalidateQueries({ queryKey: ['assignment', assignmentId] });
        },
      });
    }

    export function useDeleteAssignment(assignmentId: string, courseId: string) {
      const { getToken } = useCurrentUser();
      const qc = useQueryClient();
      return useMutation({
        mutationFn: () => deleteAssignment(assignmentId, getToken),

    export function useSubmitAssignment(assignmentId: string) {
      const { getToken } = useCurrentUser();
      const qc = useQueryClient();
      return useMutation({
        mutationFn: (circuitJson: string) => submitAssignment(assignmentId, circuitJson, getToken),
        onSuccess: () => {

          qc.invalidateQueries({ queryKey: ['assignment', assignmentId, 'submissions'] });
        },
      });
    }

    /**
     * Optimistic grade update per D-32 and 03-RESEARCH.md Pattern 4.
     * On mutate: patch the submissions list cache locally.
     * On error: roll back to previous snapshot.
     * On settled: invalidate both list and submission detail.
     */
    export function useSaveGrade(submissionId: string, assignmentId: string) {
      const { getToken } = useCurrentUser();
      const qc = useQueryClient();
      return useMutation({
        mutationFn: (input: SaveGradeInput) => saveGrade(submissionId, input, getToken),
        onMutate: async (input) => {
          await qc.cancelQueries({ queryKey: ['assignment', assignmentId, 'submissions'] });
          const previous = qc.getQueryData<SubmissionListRow[]>(
            ['assignment', assignmentId, 'submissions']
          );
          if (previous) {
            qc.setQueryData<SubmissionListRow[]>(
              ['assignment', assignmentId, 'submissions'],
              previous.map((row) =>
                row.submission_id === submissionId
                  ? { ...row, grade: input.grade, feedback: input.feedback, graded_at: Date.now() }
                  : row
              )
            );
          }
          return { previous };
        },
        onError: (_err, _input, ctx) => {
          if (ctx?.previous) {
            qc.setQueryData(['assignment', assignmentId, 'submissions'], ctx.previous);
          }
        },
        onSettled: () => {
          qc.invalidateQueries({ queryKey: ['assignment', assignmentId, 'submissions'] });
          qc.invalidateQueries({ queryKey: ['submission', submissionId] });
        },
      });
    }
